255  int64_t GetMedianTimePast() const

260 
261  const CBlockIndex* pindex = this;

264 
265  std::sort(pbegin, pend);

329  template <typename Stream, typename Operation>
330  inline void SerializationOp(Stream& s, Operation ser_action, int nType, int nVersion)
331  {
332  if (!(nType & SER_GETHASH))
333  READWRITE(VARINT(nVersion));
334 
335  READWRITE(VARINT(nHeight));
336  READWRITE(VARINT(nStatus));
337  READWRITE(VARINT(nTx));
338  if (nStatus & (BLOCK_HAVE_DATA | BLOCK_HAVE_UNDO))
339  READWRITE(VARINT(nFile));
340  if (nStatus & BLOCK_HAVE_DATA)
341  READWRITE(VARINT(nDataPos));
342  if (nStatus & BLOCK_HAVE_UNDO)
343  READWRITE(VARINT(nUndoPos));
344  READWRITE(hashAnchor);
345 
346  // block hash
347  READWRITE(hash);
348  // block header
349  READWRITE(this->nVersion);
350  READWRITE(hashPrev);
351  READWRITE(hashMerkleRoot);
352  READWRITE(hashReserved);
353  READWRITE(nTime);
354  READWRITE(nBits);
355  READWRITE(nNonce);
356  READWRITE(nSolution);
357  }
358 
359  uint256 GetBlockHash() const
360  {
361  if (hash != uint256())
362  return hash;
363  // should never really get here, keeping this as a fallback
364  CBlockHeader block;
365  block.nVersion = nVersion;
366  block.hashPrevBlock = hashPrev;
367  block.hashMerkleRoot = hashMerkleRoot;
368  block.hashReserved = hashReserved;
369  block.nTime = nTime;
370  block.nBits = nBits;
371  block.nNonce = nNonce;
372  block.nSolution = nSolution;
373  return block.GetHash();
374  }
