514  if (vAddrToSend.size() >= MAX_ADDR_TO_SEND) {
515  vAddrToSend[insecure_rand() % vAddrToSend.size()] = addr;
516  } else {
517  vAddrToSend.push_back(addr);
518  }
519  }
