255 enum {
256  // NODE_NETWORK means that the node is capable of serving the block chain. It is currently
257  // set by all Bitcoin Core nodes, and is unset by SPV clients or other peers that just want
258  // network services but don't provide them.
259  NODE_NETWORK = (1 << 0),
260  // NODE_GETUTXO means the node is capable of responding to the getutxo protocol request.
261  // Bitcoin Core does not support this but a patch set called Bitcoin XT does.
262  // See BIP 64 for details on how this is implemented.
263  NODE_GETUTXO = (1 << 1),
264 
265  // NODE_BLOOM means the node is capable and willing to handle bloom-filtered connections.
266  // Dash Core nodes used to support this by default, without advertising this bit,
267  // but no longer do as of protocol version 70201 (= NO_BLOOM_VERSION)
268  NODE_BLOOM = (1 << 2),
269 
270 
271  // Bits 24-31 are reserved for temporary experiments. Just pick a bit that
272  // isn't getting used, or one not being used much, and notify the
273  // bitcoin-development mailing list. Remember that service bits are just
274  // unauthenticated advertisements, so your code must be robust against
275  // collisions and other cases where nodes may be advertising a service they
276  // do not actually support. Other service bits should be allocated via the
277  // BIP process.
278 };
