375 inline unsigned int GetSizeOfCompactSize(uint64_t nSize)
376 {
377  if (nSize < 253)
378  return 1;
379  else if (nSize <= 0xFFFFu)
380  return 3;
381  else if (nSize <= 0xFFFFFFFFu)
382  return 5;
383  else
384  return 9;
385 }

387 template <typename Stream>
388 void WriteCompactSize(Stream& os, uint64_t nSize)
389 {
390  if (nSize < 253) {
391  ser_writedata8(os, nSize);
392  } else if (nSize <= 0xFFFFu) {
393  ser_writedata8(os, 253);
394  ser_writedata16(os, nSize);
395  } else if (nSize <= 0xFFFFFFFFu) {
396  ser_writedata8(os, 254);
397  ser_writedata32(os, nSize);
398  } else {
399  ser_writedata8(os, 255);
400  ser_writedata64(os, nSize);
401  }
402 }

404 template <typename Stream>
405 uint64_t ReadCompactSize(Stream& is)
406 {
407  uint8_t chSize = ser_readdata8(is);
408  uint64_t nSizeRet = 0;
409  if (chSize < 253) {
410  nSizeRet = chSize;
411  } else if (chSize == 253) {
412  nSizeRet = ser_readdata16(is);
413  if (nSizeRet < 253)
414  throw std::ios_base::failure("non-canonical ReadCompactSize()");
415  } else if (chSize == 254) {
416  nSizeRet = ser_readdata32(is);
417  if (nSizeRet < 0x10000u)
418  throw std::ios_base::failure("non-canonical ReadCompactSize()");
419  } else {
420  nSizeRet = ser_readdata64(is);
421  if (nSizeRet < 0x100000000ULL)
422  throw std::ios_base::failure("non-canonical ReadCompactSize()");
423  }
424  if (nSizeRet > (uint64_t)MAX_SIZE)
425  throw std::ios_base::failure("ReadCompactSize(): size too large");
426  return nSizeRet;
427 }

466 template <typename Stream, typename I>
467 void WriteVarInt(Stream& os, I n)
468 {
469  unsigned char tmp[(sizeof(n) * 8 + 6) / 7];
470  int len = 0;
471  while (true) {
472  tmp[len] = (n & 0x7F) | (len ? 0x80 : 0x00);
473  if (n <= 0x7F)
474  break;
475  n = (n >> 7) - 1;
476  len++;
477  }
478  do {
479  ser_writedata8(os, tmp[len]);
480  } while (len--);
481 }

483 template <typename Stream, typename I>
484 I ReadVarInt(Stream& is)
485 {
486  I n = 0;
487  while (true) {
488  unsigned char chData = ser_readdata8(is);
489  n = (n << 7) | (chData & 0x7F);
490  if (chData & 0x80)
491  n++;
492  else
493  return n;
494  }
495 }
