270 template <typename Callable>
271 void TraceThread(const char* name, Callable func)
272 {
273  std::string s = strprintf("zcash-%s", name);
274  RenameThread(s.c_str());
275  try {
276  LogPrintf("%s thread start\n", name);
277  func();
278  LogPrintf("%s thread exit\n", name);
279  } catch (const boost::thread_interrupted&) {
280  LogPrintf("%s thread interrupt\n", name);
281  throw;
282  } catch (const std::exception& e) {
283  PrintExceptionContinue(&e, name);
284  throw;
285  } catch (...) {
286  PrintExceptionContinue(NULL, name);
287  throw;
288  }
289 }
